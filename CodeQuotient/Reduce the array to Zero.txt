class Result {
  static ArrayList<Integer> reduceArray(ArrayList<Integer> arr) {
    
    Collections.sort(arr);
    int s=arr.get(0);
    int n=arr.size();
    int count = 1;
    ArrayList<Integer> ans = new ArrayList<>();
    ans.add(n);
    for(int i=1; i<arr.size(); i++)
    {
      if(arr.get(i) == s)
      {
        count++;
      }
      if(arr.get(i) > s)
      {
        s= arr.get(i);
        ans.add(n-count);
        n=n-count;
        count = 1;
      }
    }
    return ans;
  }
}
